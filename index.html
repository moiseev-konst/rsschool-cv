function indexEqualsValue(a) {
    let result = -1
    let start = 0
    let end = a.length - 1
    while (end - start >= 0) {
        let middle = (end - start) >> 1
        middle += start
        if (a[middle] == middle) {
             result = middle
        }
        if (a[middle] >= middle) {
            end = middle - 1
        } else {
            start = middle + 1
        }
    }
    return result
}
